💡 使用例

アイテム登録

INSERT INTO shop_items 
(item_name, price, effect_type, effect_value, description) 
VALUES ('パワードリンク', 100, 'power', 10, 'パワーを10上昇させるドリンク');

アイテム一覧取得

SELECT item_name, price, effect_type, effect_value 
FROM shop_items 
ORDER BY effect_type, price;

カテゴリ別アイテム取得

SELECT * FROM shop_items 
WHERE effect_type = 'power' 
ORDER BY effect_value DESC;

価格帯別アイテム取得

SELECT * FROM shop_items 
WHERE price BETWEEN 100 AND 200 
ORDER BY price;

アイテム統計

SELECT effect_type, 
       COUNT(*) as item_count,
       AVG(price) as avg_price,
       MAX(effect_value) as max_effect
FROM shop_items 
GROUP BY effect_type;

コストパフォーマンス順

SELECT item_name, price, effect_value,
       (effect_value / price) as cost_performance
FROM shop_items 
ORDER BY cost_performance DESC;
